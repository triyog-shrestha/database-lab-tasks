1. Show all databases in the cluster

show dbs

2. Use a database

use <db_name>

3. Insert multiple documents into a collection

db.students.insertMany([
{ name: "Bob",   age: 30, section: "A" },
{ name: "Carol", age: 22, section: "B" },
{ name: "Dave",  age: 28, section: "A" }
])

4. Insert a single document

db.students.insertOne({ name: "Bron", age: 42, section: "B" })

5. Show all documents

db.students.find().pretty()

6. Show students older than 20

db.students.find({ age: { $gt: 20 } }).pretty()

7. Show students in section A

db.students.find({ section: "A" }).pretty()

8. Show students named Bron

db.students.find({ name: "Bron" }).pretty()

9. Update one document

db.students.updateOne({ name: "Bob" }, { $set: { section: "C" } })

10. Update many documents

db.students.updateMany({ age: { $lt: 30 } }, { $set: { grade: "D" } })

11. Delete one document

db.students.deleteOne({ name: "Bob" })

12. Delete many documents

db.students.deleteMany({ age: { $lt: 30 } })

13. Count all documents in a collection

db.students.countDocuments()

14. Count documents matching a filter

db.students.countDocuments({ section: "A" })

15. Sort in ascending order

db.students.find().sort({ age: 1 }).pretty()

16. Sort in descending order

db.students.find().sort({ age: -1 }).pretty()

17. Add a new field to all documents

db.students.updateMany({}, { $set: { city: "Kathmandu" } })

18. Delete students older than 21

db.students.deleteMany({ age: { $gt: 21 } })

19. Delete students between age 17 and 20

db.students.deleteMany({
  age: { $gte: 17, $lte: 20 }
})
